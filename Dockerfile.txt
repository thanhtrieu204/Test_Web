# Sử dụng môi trường Tomcat 9 chạy trên Java 11 (hoặc đổi thành tomcat:10 nếu bạn dùng bản mới)
FROM tomcat:9.0-jdk11-openjdk

# Xóa các file mặc định của Tomcat để tránh nặng máy
RUN rm -rf /usr/local/tomcat/webapps/*

# Copy file .war từ thư mục dist của NetBeans vào thư mục chạy của Tomcat trên Cloud
# CHÚ Ý: Thay thế "Ten_File_Cua_Ban.war" bằng tên file thực tế trong thư mục dist của bạn
COPY dist/Ten_File_Cua_Ban.war /usr/local/tomcat/webapps/ROOT.war

# Mở cổng 8080 để Render có thể kết nối
EXPOSE 8080

# Lệnh khởi chạy Tomcat khi Server khởi động
CMD ["catalina.sh", "run"]
